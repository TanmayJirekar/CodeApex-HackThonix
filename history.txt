# Route to get health data by ID
@app.route("/get/<int:record_id>", methods=["GET"])
def get_health_data_by_id(record_id):
conn = get_db_connection()
cursor = conn.cursor(dictionary=True)
cursor.execute("SELECT * FROM health_data WHERE id = %s", (record_id,))
data = cursor.fetchone()
cursor.close()
conn.close()

if data:
return jsonify(data)
